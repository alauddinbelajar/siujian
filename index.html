const SHEET_NAME_SOAL = 'SOAL'; const SHEET_NAME_HASIL = 'HASIL'; function doGet(e) { const action = e.parameter.action; let result; if (action === 'getSoal') result = getSoalData(); else if (action === 'getHasil') result = getHasilData(); else if (action === 'saveHasil' && e.parameter.payload) { try { const data = JSON.parse(decodeURIComponent(e.parameter.payload)); result = saveHasilData(data); } catch(err) { result = {status:'error', msg:err.message}; } } else { result = {error:'Unknown action'}; } const json = JSON.stringify(result); return ContentService.createTextOutput(json).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { try { const data = JSON.parse(e.postData.contents); if (data.action === 'saveSoal') return saveSoal(data.soal); if (data.action === 'saveHasil') { const r = saveHasilData(data); return ContentService.createTextOutput(JSON.stringify(r)).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({status:'error',msg:'Unknown action'})).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:'error',msg:err.message})).setMimeType(ContentService.MimeType.JSON); } } function getSoalData() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME_SOAL); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME_SOAL); setupSoalHeader(sheet); } const data = sheet.getDataRange().getValues(); if (data.length <= 1) return {soal:[]}; const headers = data[0]; const soal = data.slice(1).filter(r=>r[0]).map(r => { const obj = {}; headers.forEach((h,i) => obj[h] = r[i] !== undefined ? String(r[i]) : ''); obj.poin = parseFloat(String(obj.poin).replace(',','.')) || 1; return obj; }); return {soal}; } function saveSoal(soalArr) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME_SOAL); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME_SOAL); } sheet.clearContents(); const headers = ['tipe','soal','opsiA','opsiB','opsiC','opsiD','opsiE','jawaban','poin','gambarSoal','gambarOpsiA','gambarOpsiB','gambarOpsiC','gambarOpsiD','gambarOpsiE']; sheet.appendRow(headers); soalArr.forEach(s => sheet.appendRow(headers.map(h => s[h]||''))); return {status:'ok'}; } function saveHasilData(data) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME_HASIL); const analisis = data.analisis || []; const jawabanHeaders = analisis.map((_,i) => 'Jawaban'+(i+1)); const poinHeaders = analisis.map((_,i) => 'Poin'+(i+1)); const baseHeaders = ['Nama','Kelas','Ujian','Nilai','Total','Durasi','Waktu']; const allHeaders = [...baseHeaders, ...jawabanHeaders, ...poinHeaders]; if (!sheet) { sheet = ss.insertSheet(SHEET_NAME_HASIL); sheet.appendRow(allHeaders); formatHeader_(sheet, allHeaders.length); } else if (sheet.getLastRow() === 0) { sheet.appendRow(allHeaders); formatHeader_(sheet, allHeaders.length); } else { const existingCols = sheet.getLastColumn(); if (existingCols < allHeaders.length) { sheet.getRange(1,1,1,allHeaders.length).setValues([allHeaders]); formatHeader_(sheet, allHeaders.length); } } const jawabanData = analisis.map(a => a.jawaban || '(kosong)'); const poinData = analisis.map(a => { const p = a.benar ? (parseFloat(String(a.poin||1).replace(',','.')) || 1) : 0; return String(p).replace('.',','); }); const row = [ data.name||'', data.kelas||'', data.ujian||'', data.score||0, data.totalQ||0, data.durasi||'', data.submitTime||'', ...jawabanData, ...poinData ]; sheet.appendRow(row); return {status:'ok'}; } function formatHeader_(sheet, numCols) { const range = sheet.getRange(1,1,1,numCols); range.setBackground('#1A3A5C'); range.setFontColor('#FFFFFF'); range.setFontWeight('bold'); sheet.setFrozenRows(1); } function setupSoalHeader(sheet) { sheet.appendRow(['tipe','soal','opsiA','opsiB','opsiC','opsiD','opsiE','jawaban','poin','gambarSoal','gambarOpsiA','gambarOpsiB','gambarOpsiC','gambarOpsiD','gambarOpsiE']); } function getHasilData() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME_HASIL); if (!sheet || sheet.getLastRow() <= 1) return {hasil:[]}; const data = sheet.getDataRange().getValues(); const headers = data[0].map(h=>String(h).toLowerCase().trim()); const hasil = data.slice(1).filter(r=>r[0]).map(r => { const obj = {}; headers.forEach((h,i) => { obj[h] = r[i] !== undefined ? String(r[i]) : ''; }); return { name: obj['nama']||obj['name']||'', kelas: obj['kelas']||'', ujian: obj['ujian']||'', score: parseFloat(String(obj['nilai']||obj['score']||'0').replace(',','.'))||0, totalQ: parseInt(obj['total']||obj['totalq'])||0, durasi: obj['durasi']||'', submitTime: obj['waktu']||obj['submittime']||'' }; }); return {hasil}; }
